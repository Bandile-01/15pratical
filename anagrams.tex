PrintWriter out = new PrintWriter(new FileWriter("theAnagrams.tex"));

out.println("\\begin{itemize}");

for (String key : D.keySet()) {

    ArrayList<String> list = D.get(key);

    if (list.size() >= 2) {

        out.print("\\item " + key + " : ");

        for (int i = 0; i < list.size(); i++) {

            out.print(list.get(i));

            if (i < list.size() - 1) {
                out.print(", ");
            }
        }

        out.println();
    }
}

out.println("\\end{itemize}");
out.close();
